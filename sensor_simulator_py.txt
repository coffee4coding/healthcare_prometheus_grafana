




from prometheus_client import start_http_server, Gauge
import random
import time

# Define Prometheus metric
sensor_value_gauge = Gauge('sensor_value', 'Simulated Sensor Value')

# Start HTTP server on port 5000
start_http_server(5000)

while True:
    value = random.randint(50, 100)
    sensor_value_gauge.set(value)
    time.sleep(5)
